class Solution { public: int minCostClimbingStairs(vector& cost) { int total = cost.size(); int dp1[1005],dp2[1005]; dp1[0] = 0; dp2[0] = 0; dp2[1] = 0; if (total == 0||total == 1) return 0; if (total == 2) return cost[0]; for (int i = 1; i <= total ; ++i) { if (i == 1) { dp1[i] = cost[0]; continue; } dp1[i] = min(dp1[i-1] +cost[i-1],dp1[i-2]+cost[i-2]); } for (int i = 2; i <= total ; ++i) { dp2[i] = min(dp2[i-1] +cost[i-1],dp2[i-2]+cost[i-2]); } return min(dp1[total],dp2[total]); } };
